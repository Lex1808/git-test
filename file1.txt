my_list = []
swapped = True
num = int(input("Скільки елементів потрібно відсортувати: "))

for i in range(num):
   val = float(input("Введіть елемент списку: "))
   my_list.append(val)

while swapped:
   swapped = False
   for i in range(len(my_list) - 1):
     if my_list[i] > my_list[i + 1]:
       swapped = True
       my_list[i], my_list[i + 1] = my_list[i + 1], my_list[i]

print("\nВідсортовано:")
print(my_list)
